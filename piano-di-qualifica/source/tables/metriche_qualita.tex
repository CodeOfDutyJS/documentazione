\newcommand*\metricstable{}
\foreach \x [count=\nj] in \productquality
{
    \foreach \y [count=\ni] in \x
    {
        \ifnum\ni=4
            \xappto\metricstable{\y}
            \gappto\metricstable{\\}
            \gappto\metricstable{\hline}
        \else\ifnum\ni=5
            \xappto\metricstable{\noexpand\multicolumn{4}{| l |}{NOTE: \y}}
            \gappto\metricstable{\\}
            \gappto\metricstable{\hline}
        \else
            \xappto\metricstable{\y&}
        \fi\fi
    }
}
