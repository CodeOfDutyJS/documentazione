    \begin{itemize}
        \item \textbf{An}: Periodo di analisi, ovvero dal 26-11-2020 al 10-01-2021
        \item \textbf{CC}: periodo di consolidamento dei requisiti, ovvero dal 12-01-2021 al 18-01-2021
