\newcommand*\risktable{}

{
    \foreach \y [count=\ni] in \x
    {
        \ifnum\ni=3
            \xappto\metricstable{\y}
            \gappto\metricstable{\\}
            \gappto\metricstable{\hline}
        \else
            \xappto\metricstable{\y & }
        \fi
    }
}
