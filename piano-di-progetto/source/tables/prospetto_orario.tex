\newcommand*\hourlysummary{}
\foreach \x [count=\nj] in \hourlycontent
{
    \foreach \y [count=\ni] in \x
    {
        \ifnum\ni=1
            \xappto\hourlysummary{\noexpand\textbf{\y}&}
        \else\ifnum\ni=8
            \xappto\hourlysummary{\y}
            \gappto\hourlysummary{\\}
            \gappto\hourlysummary{\hline}
        \else
            \xappto\hourlysummary{\y&}
        \fi\fi
    }
}
